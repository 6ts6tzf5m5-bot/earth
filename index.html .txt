    <div class="stat-item"><div class="dot" style="background: #00ff00; box-shadow: 0 0 8px #00ff00;"></div> ISS (人工衛星)</div>

  <script>
    // 1. 基本設定（夜景が綺麗に見えるように設定）
    const viewer = new Cesium.Viewer('cesiumContainer', {
      terrainProvider: Cesium.createWorldTerrain(),
      shouldAnimate: true,
      animation: false, // 下部の時計パネルを非表示
      timeline: false,  // タイムラインを非表示
      baseLayerPicker: false
    });

    // 暗い宇宙に映えるよう、夜間照明（自動）を設定
    viewer.scene.globe.enableLighting = true;

    // --- 共通設定：航跡を持つオブジェクトの生成 ---
    function addLiveObject(id, name, color) {
      return viewer.entities.add({
        id: id,
        name: name,
        position: new Cesium.SampledPositionProperty(),
        path: {
          resolution: 1,
          material: new Cesium.PolylineGlowMaterialProperty({
            glowPower: 0.2,
            color: color
          }),
          width: 2,
          trailTime: 600 // 10分間の航跡を表示
        },
        point: { pixelSize: 7, color: color, outlineColor: Cesium.Color.BLACK, outlineWidth: 2 },
        label: { 
          text: name, 
          font: '10pt sans-serif', 
          pixelOffset: new Cesium.Cartesian2(0, -15),
          showBackground: true,
          backgroundColor: new Cesium.Color(0, 0, 0, 0.6)
        }
      });
    }

    // --- 2. 衛星（ISS）のシミュレーション ---
    const iss = addLiveObject('iss-unique', 'ISS (Satellite)', Cesium.Color.LIME);
    let satelliteAngle = 0;

    viewer.clock.onTick.addEventListener(() => {
      satelliteAngle += 0.03;
      const time = viewer.clock.currentTime;
      const lon = (satelliteAngle * 2.2) % 360 - 180;
      const lat = 51.6 * Math.sin(satelliteAngle * 0.5); // ISSの傾斜角に近い軌道
      const pos = Cesium.Cartesian3.fromDegrees(lon, lat, 420000);
      iss.position.addSample(time, pos);
    });

    // --- 3. 飛行機データのリアルタイム同期 ---
    async function syncFlights() {
      const msg = document.getElementById('status-msg');
      msg.innerText = '航空データを更新しています...';
      
      try {
        const response = await fetch('https://opensky-network.org/api/states/all');
        const data = await response.json();
        const time = viewer.clock.currentTime;

        // 負荷軽減のため、最初の40機に絞る
        data.states.slice(0, 40).forEach(f => {
          const id = "plane_" + f[0];
          const callsign = f[1].trim() || "FLIGHT";
          const lon = f[5];
          const lat = f[6];
          const alt = f[7] || 10000;

          if (lon && lat) {
            let plane = viewer.entities.getById(id);
            if (!plane) {
              plane = addLiveObject(id, callsign, Cesium.Color.ORANGE);
            }
            const pos = Cesium.Cartesian3.fromDegrees(lon, lat, alt);
            plane.position.addSample(time, pos);
          }
        });
        msg.innerText = 'LIVE: 航空データ同期完了';
      } catch (err) {
        msg.innerText = 'API待機中（数秒後に再試行）';
      }
    }

    // 15秒ごとにフライト情報を更新
    setInterval(syncFlights, 15000);
    syncFlights();

  </script>
